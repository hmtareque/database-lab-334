create table output_tab(
	position number,
	num number,
	result varchar2(20),
	summation number
);
commit;

declare
  tot number;
  cursor val_cursor is
    select * from input_tab;
  cursor cal_cursor is
    select * from input_tab;
    
begin
  for val in val_cursor
    loop
      if( val.num mod 2 = 0 ) then
        tot := 0;
        
        for cal in cal_cursor
        loop
          if( cal.num mod 2 = 0 and cal.num < val.num ) then
            tot := tot + cal.num;
          end if;    
        end loop;
        
        insert into output_tab values(val.position,val.num,'EVEN.',tot);
      	commit;
      
      else
        tot := 0;
        
        for cal in cal_cursor
        loop
          if( cal.num mod 2 = 1 and cal.num < val.num ) then
            tot := tot + cal.num; 
          end if;   
        end loop;
        
        insert into output_tab values(val.position,val.num,'ODD.',tot);
      	commit;
      end if;
      
    end loop;
end;
